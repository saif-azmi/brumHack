create a new branch and switch to it
	git checkout -b [branch name]
	
	commits then push to this branch as default

switch back to master (commit everything first!)
	git checkout master

when you complete work on a branch
	//add the branch changes to master (checkout master first)
	git merge [branch name]
	//delete the temp branch (changes are now part of master so  this is safe)
	git branch -d [branch name]
